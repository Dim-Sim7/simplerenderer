#include "geometry.h"

#include "tgaimage.h"

constexpr int width  = 128;
constexpr int height = 128;

constexpr TGAColor white   = {255, 255, 255, 255}; // attention, BGRA order
constexpr TGAColor green   = {  0, 255,   0, 255};
constexpr TGAColor red     = {  0,   0, 255, 255};
constexpr TGAColor blue    = {255, 128,  64, 255};
constexpr TGAColor yellow  = {  0, 200, 255, 255};

void line(int ax, int ay, int bx, int by, TGAImage &framebuffer, TGAColor color) {
    bool steep = std::abs(ax-bx) < std::abs(ay-by);
    if (steep) { // if the line is steep, we transpose the image
        std::swap(ax, ay);
        std::swap(bx, by);
    }
    if (ax>bx) { // make it left−to−right
        std::swap(ax, bx);
        std::swap(ay, by);
    }
    int y = ay;
    int ierror = 0;
    for (int x=ax; x<=bx; x++) {
        if (steep) // if transposed, de−transpose
            framebuffer.set(y, x, color);
        else
            framebuffer.set(x, y, color);
        ierror += 2 * std::abs(by-ay);
        if (ierror > bx - ax) {
            y += by > ay ? 1 : -1;
            ierror -= 2 * (bx-ax);
        }
    }
}
//Scanline algo
//define ymin and ymax of all vertices
//for r in ymin to ymax - find end points of r intersected with triangle (lEnd and rEnd)
//for c in ceiling(ymin) to floor(ymax) - render pixels

struct Point {
    int x, y;
};

void scanlineRender(Point p1, Point p2, Point p3, TGAImage &framebuffer, TGAColor color) {
    int yMin = std::ceil(std::min({p1.y, p2.y, p3.y}));
    int yMax = std::floor(std::max({p1.y, p2.y, p3.y}));

    //define triangle edges
    Point edges[3][2] = { {p1, p2}, {p2, p3}, {p3, p1} };



    //loop through min and max values of y (in relation to triangle)
    //determine where the triangle intersects at x plane
    for (int y = yMin; y <+ yMax; y++) {
        std::vector<double> xIntersections;

        //check each edge for intersection
        for (int e = 0; e < 3; e++) {
            Point A = edges[e][0];
            Point B = edges[e][1];

            //Skip horizontal edges
            if (A.y == B.y) continue;

            //check if scanline intersects the edge
            //y is between point A and Point B
            if ((y >= A.y && y <= B.y) || (y >= B.y && y <= A.y)) {
                //linear interpolation to find x
                // x = A.x + ((B.x - A.x) * (y - A.y)) / (B.y - A.y) 
                //(B.x - A.x) / (B.y - A.y) is the slope of the edge in x per unit y
                //(y - A.y) is how far the scanline is vertically from Point A
                //multiply previous two together to get x offset and add to A.x to get exact x coord of intersection
                float x = A.x + (float)(B.x - A.x) * (y - A.y) / (B.y - A.y);
                xIntersections.push_back(x);
            }
        }
        
        //need exactly two intersections for a triangle scan line
        if (xIntersections.size() >= 2) {
            std::sort(xIntersections.begin(), xIntersections.end());
            int leftEnd = ceil(xIntersections[0]);
            int rightEnd = floor(xIntersections[1]);
        
            //draw horizontal line between intersections
            for (int x = leftEnd; x < rightEnd; x++) {
                framebuffer.set(x, y, color);
            }
        }
        
    }

}

void triangle(int ax, int ay, int bx, int by, int cx, int cy, TGAImage &framebuffer, TGAColor color) {

    Point p1 = {ax, ay};
    Point p2 = {bx, by};
    Point p3 = {cx, cy};
    scanlineRender(p1, p2, p3, framebuffer, color);
}

int main(int argc, char** argv) {
    TGAImage framebuffer(width, height, TGAImage::RGB);

    triangle(  7, 45, 35, 100, 45,  60, framebuffer, red);
    triangle(120, 35, 90,   5, 45, 110, framebuffer, white);
    triangle(115, 83, 80,  90, 85, 120, framebuffer, green);
    framebuffer.write_tga_file("framebuffer.tga");
    return 0;
}
